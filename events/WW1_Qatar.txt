	}
}

# The Question of Qatari Suzerainty
country_event = {
	id = ww1_qatar.2
	title = ww1_qatar.2.t
	desc = ww1_qatar.2.d
	picture = GFX_report_event_tur_turkish_soldiers
	fire_only_once = yes
	trigger = {
		original_tag = TUR
		QAT = {
			is_subject_of = ROOT
			NOT = {
				is_neighbor_of_subject = yes #Al-Hasa
			}
		}
	}
	mean_time_to_happen = { days = 40 }
	option = {
		name = ww1_qatar.2.a
		end_puppet = QAT
		QAT = {
			clr_country_flag = protectorate
			if = {
				limit = {
					is_in_faction_with = ROOT
				}
				leave_faction = yes
			}
			give_military_access = ROOT
			every_other_country = {
				limit = {
					has_war_with = ROOT
					has_war_with = PREV
				}
				white_peace = PREV
			}
		}
	}
}
